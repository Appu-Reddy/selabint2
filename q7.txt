updatedoc.js

const Student = require('../model/student.js');
module.exports = async (req, res) => {
    try {
        const updateData = {};
        if (req.body.name) updateData.name = req.body.name;
        if (req.body.id) updateData.id = req.body.id;
        if (req.body.course) updateData.course = req.body.course;
        if (req.body.branch) updateData.branch = req.body.branch;
        const result = await Student.updateOne(
            { name: req.body.cname },
            { $set: updateData }
        );
        console.log(result);
        const message = result.modifiedCount > 0 
            ? 'Student Details Updated Successfully' 
            : 'No Matching Student Found or No Changes Made';

        res.render('update', { status: message });
    }
    catch (error) {
        console.error(error);
        res.render('update', { status: 'Cannot Update Student Details. Please Try Again' });
    }
};

student.js

const mongoose = require('mongoose');
const studentSchema = new mongoose.Schema({
    name: {
        type: String,
        required: true,
        unique: true
    },
    id: {
        type: String,
        required: true,
    },
    course: {
        type: String,
        required: true
    },
    branch: {
        type: String,
        required: true
    }
});
module.exports = mongoose.model('Student',studentSchema);

app.js

const express = require('express');
const app = express();
const path = require('path');
const mongoose = require('mongoose');
const searchStudentController = require('./controller/searchStudent');
const fetchController = require('./controller/searchdb');
const newStudentController = require('./controller/newCustomer');  // shows registration form
const storeStudentController = require('./controller/storeCustomer'); // handles registration POST
const updateStudentController = require('./controller/updateCustomer'); // shows update form
const updatedocController = require('./controller/updatedoc'); // handles update POST

app.use(express.json());
app.use(express.urlencoded({ extended: true }));
app.use(express.static(path.join(__dirname, 'public'))); // if you have CSS/JS
app.set('views', path.join(__dirname, 'view'));
app.set('view engine', 'ejs');

mongoose.connect('mongodb://localhost/student')
    .then(() => console.log('Connected to database...'))
    .catch(err => console.error('Database connection error:', err));

app.get('/', (req, res) => {
    res.render('index');
});

app.get('/page/register', newStudentController);
app.post('/student/store', storeStudentController);
app.get('/page/search', searchStudentController);
app.post('/student/fetch', fetchController);
app.get('/page/update', updateStudentController);
app.post('/student/update', updatedocController);

const PORT = 3000;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}...`);
});
